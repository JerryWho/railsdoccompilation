240:         if @options[:interactive]
241:           require 'sass'
242:           require 'sass/repl'
243:           ::Sass::Repl.new(@options).run
244:           return
245:         end
246: 
247:         super
248:         input = @options[:input]
249:         output = @options[:output]
250: 
251:         tree =
252:           if input.is_a?(File) && !@options[:check_syntax]
253:             ::Sass::Files.tree_for(input.path, @options[:for_engine])
254:           else
255:             # We don't need to do any special handling of @options[:check_syntax] here,
256:             # because the Sass syntax checking happens alongside evaluation
257:             # and evaluation doesn't actually evaluate any code anyway.
258:             ::Sass::Engine.new(input.read(), @options[:for_engine]).to_tree
259:           end
260: 
261:         input.close() if input.is_a?(File)
262: 
263:         output.write(tree.render)
264:         output.close() if output.is_a? File
265:       rescue ::Sass::SyntaxError => e
266:         raise e if @options[:trace]
267:         raise "Syntax error on line #{get_line e}: #{e.message}"
268:       end
