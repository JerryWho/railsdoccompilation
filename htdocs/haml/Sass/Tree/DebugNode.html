def _perform(environment)
  res = @expr.perform(environment)
  res = res.value if res.is_a?(Sass::Script::String)
  if filename
    $stderr.puts "#{filename}:#{line} DEBUG: #{res}"
  else
    $stderr.puts "Line #{line} DEBUG: #{res}"
  end
  []
end
