def to_feed(rss)
  textInput = Rss::Channel::TextInput.new
  set = setup_values(textInput)
  if set
    rss.channel.textInput = textInput
    set_parent(textInput, rss.channel)
    setup_other_elements(rss, textInput)
  end
end
