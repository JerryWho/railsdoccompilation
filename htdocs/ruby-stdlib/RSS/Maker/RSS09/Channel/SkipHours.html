def to_feed(rss, channel)
  unless @hours.empty?
    skipHours = Rss::Channel::SkipHours.new
    channel.skipHours = skipHours
    set_parent(skipHours, channel)
    @hours.each do |hour|
      hour.to_feed(rss, skipHours.hours)
    end
  end
end
