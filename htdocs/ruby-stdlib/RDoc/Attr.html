def inspect # :nodoc:
  attr = case rw
         when 'RW' then :attr_accessor
         when 'R'  then :attr_reader
         when 'W'  then :attr_writer
         else
             " (#{rw})"
         end

    "#<%s:0x%x %s.%s :%s>" % [
      self.class, object_id,
      parent_name, attr, @name,
    ]
end

def marshal_load array

  @full_name  = array[2]
