def accept visitor
  visitor.accept_list_start self

  @items.each do |item|
    item.accept visitor
  end

  visitor.accept_list_end self
end

def pretty_print q # :nodoc:
  q.group 2, "[list: #{@type} ", ']' do
    q.seplist @items do |item|
      q.pp item
    end
  end
end
