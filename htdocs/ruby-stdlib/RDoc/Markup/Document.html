def << part
  case part
  when RDoc::Markup::Document then
    unless part.empty? then
      parts.push(*part.parts)
      parts << RDoc::Markup::BlankLine.new
    end
  when String then
    raise ArgumentError,
          "expected RDoc::Markup::Document and friends, got String" unless
      part.empty?
  else
    parts << part
  end
end

def pretty_print q # :nodoc:

    end
  end
end
