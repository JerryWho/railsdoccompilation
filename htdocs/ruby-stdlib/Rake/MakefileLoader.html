def load(fn)
  lines = open(fn) {|mf| mf.read}
  lines.gsub!(/\\ /, SPACE_MARK)
  lines.gsub!(/#[^\n]*\n/m, "")
  lines.gsub!(/\\\n/, ' ')
  lines.each_line do |line|
    process_line(line)
  end
end
