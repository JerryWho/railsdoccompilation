static VALUE get_c1(VALUE self)
{
  xmlElementContentPtr elem;
  Data_Get_Struct(self, xmlElementContent, elem);

  if(!elem->c1) return Qnil;
  return Nokogiri_wrap_element_content(rb_iv_get(self, "@document"), elem->c1);
}
