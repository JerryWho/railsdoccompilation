file

/*
 * call-seq:
 *  file
 *
 * Get the filename for the error
 */
static VALUE file(VALUE self)
{
  xmlErrorPtr error;
  Data_Get_Struct(self, xmlError, error);
  if(error->file)
    return NOKOGIRI_STR_NEW2(error->file, "UTF-8");

  return Qnil;
}
