75:       def snippet_text(step_keyword, step_name, multiline_arg_class)
76:         escaped = Regexp.escape(step_name).gsub('\ ', ' ').gsub('/', '\/')
77:         escaped = escaped.gsub(PARAM_PATTERN, ESCAPED_PARAM_PATTERN)
78: 
79:         n = 0
80:         block_args = escaped.scan(ESCAPED_PARAM_PATTERN).map do |a|
81:           n += 1
82:           "arg#{n}"
83:         end
84:         block_args << multiline_arg_class.default_arg_name unless multiline_arg_class.nil?
85:         block_arg_string = block_args.empty? ? "" : " |#{block_args.join(", ")}|"
86:         multiline_class_comment = ""
87:         if(multiline_arg_class == Ast::Table)
88:           multiline_class_comment = "# #{multiline_arg_class.default_arg_name} is a #{multiline_arg_class.to_s}\n  "
89:         end
90: 
91:         "#{step_keyword} /^#{escaped}$/ do#{block_arg_string}\n  #{multiline_class_comment}pending # express the regexp above with the code you wish you had\nend"
