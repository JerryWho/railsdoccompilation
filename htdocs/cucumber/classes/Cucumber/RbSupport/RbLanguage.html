70:       def snippet_text(step_keyword, step_name, multiline_arg_class)
71:         escaped = Regexp.escape(step_name).gsub('\ ', ' ').gsub('/', '\/')
72:         escaped = escaped.gsub(PARAM_PATTERN, ESCAPED_PARAM_PATTERN)
73: 
74:         n = 0
75:         block_args = escaped.scan(ESCAPED_PARAM_PATTERN).map do |a|
76:           n += 1
77:           "arg#{n}"
78:         end
79:         block_args << multiline_arg_class.default_arg_name unless multiline_arg_class.nil?
80:         block_arg_string = block_args.empty? ? "" : " |#{block_args.join(", ")}|"
81:         multiline_class_comment = ""
82:         if(multiline_arg_class == Ast::Table)
83:           multiline_class_comment = "# #{multiline_arg_class.default_arg_name} is a #{multiline_arg_class.to_s}\n  "
84:         end
85: 
86:         "#{step_keyword} /^#{escaped}$/ do#{block_arg_string}\n  #{multiline_class_comment}pending\nend"
87:       end
