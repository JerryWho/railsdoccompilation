74:       def snippet_text(step_keyword, step_name, multiline_arg_class)
75:         escaped = Regexp.escape(step_name).gsub('\ ', ' ').gsub('/', '\/')
76:         escaped = escaped.gsub(PARAM_PATTERN, ESCAPED_PARAM_PATTERN)
77: 
78:         n = 0
79:         block_args = escaped.scan(ESCAPED_PARAM_PATTERN).map do |a|
80:           n += 1
81:           "arg#{n}"
82:         end
83:         block_args << multiline_arg_class.default_arg_name unless multiline_arg_class.nil?
84:         block_arg_string = block_args.empty? ? "" : " |#{block_args.join(", ")}|"
85:         multiline_class_comment = ""
86:         if(multiline_arg_class == Ast::Table)
87:           multiline_class_comment = "# #{multiline_arg_class.default_arg_name} is a #{multiline_arg_class.to_s}\n  "
88:         end
89: 
90:         "#{step_keyword} /^#{escaped}$/ do#{block_arg_string}\n  #{multiline_class_comment}pending\nend"
91:       end
