371:         def get_step_count(features)
372:           count = 0
373:           features = features.instance_variable_get("@features")
374:           features.each do |feature|
375:             #get background steps
376:             if feature.instance_variable_get("@background")
377:               background = feature.instance_variable_get("@background").instance_variable_get("@steps").instance_variable_get("@steps")

382:               #get steps

388:               unless examples.nil?

393:               end
394: 
395:               #get multiline step tables

397:                 multi_arg = step.instance_variable_get("@multiline_arg")
398:                 next if multi_arg.nil?
399:                 matrix = multi_arg.instance_variable_get("@cell_matrix")
400:                 count += matrix.size unless matrix.nil?
401:               end
402:             end
403:           end
404:           return count
