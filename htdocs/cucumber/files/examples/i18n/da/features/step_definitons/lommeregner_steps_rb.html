begin require 'rspec/expectations'; rescue LoadError; require 'spec/expectations'; end

  @calc = Lommeregner.new
end

end

  @calc.push n.to_i
end

When 'jeg lægger sammen' do
  @result = @calc.add
end

  @result.should == result.to_i
