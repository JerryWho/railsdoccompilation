require 'rubygems' unless ENV['NO_RUBYGEMS']
require 'flexmock/rspec'

module Spec
  module Adapters
    module MockFramework
      include FlexMock::MockContainer
      def setup_mocks_for_rspec
        # No setup required
      end
      def verify_mocks_for_rspec
        flexmock_verify
      end
      def teardown_mocks_for_rspec
        flexmock_close
      end
    end
  end
end
