[ show source ]

     # File railties/test/boot_test.rb, line 162
162:   def test_should_allow_advanced_rubygems_version_specifications
163:     # See http://rubygems.org/read/chapter/16
164:     assert_equal "=1.2.3", parse("RAILS_GEM_VERSION = '=1.2.3'") # equal sign
165:     assert_equal "= 1.2.3", parse("RAILS_GEM_VERSION = '= 1.2.3'") # with space
166:     assert_equal "!=1.2.3", parse("RAILS_GEM_VERSION = '!=1.2.3'") # not equal
167:     assert_equal ">1.2.3", parse("RAILS_GEM_VERSION = '>1.2.3'") # greater than
168:     assert_equal "<1.2.3", parse("RAILS_GEM_VERSION = '<1.2.3'") # less than
169:     assert_equal ">=1.2.3", parse("RAILS_GEM_VERSION = '>=1.2.3'") # greater than or equal
170:     assert_equal "<=1.2.3", parse("RAILS_GEM_VERSION = '<=1.2.3'") # less than or equal
171:     assert_equal "~>1.2.3.0", parse("RAILS_GEM_VERSION = '~>1.2.3.0'") # approximately greater than
172:   end
