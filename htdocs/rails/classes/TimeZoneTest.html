114:   def test_local_enforces_spring_dst_rules
115:     zone = ActiveSupport::TimeZone['Eastern Time (US & Canada)']
116:     twz = zone.local(2006,4,2,1,59,59) # 1 second before DST start
117:     assert_equal Time.utc(2006,4,2,1,59,59), twz.time

119:     assert_equal false, twz.dst?

122:     assert_equal Time.utc(2006,4,2,3), twz2.time # twz is created for 3AM
123:     assert_equal Time.utc(2006,4,2,7), twz2.utc
124:     assert_equal true, twz2.dst?
125:     assert_equal 'EDT', twz2.zone
126:     twz3 = zone.local(2006,4,2,2,30) # 2:30AM does not exist because at 2AM, time springs forward to 3AM
127:     assert_equal Time.utc(2006,4,2,3,30), twz3.time # twz is created for 3:30AM
128:     assert_equal Time.utc(2006,4,2,7,30), twz3.utc
