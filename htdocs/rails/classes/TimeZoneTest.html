100:   def test_local_enforces_spring_dst_rules
101:     zone = ActiveSupport::TimeZone['Eastern Time (US & Canada)']
102:     twz = zone.local(2006,4,2,1,59,59) # 1 second before DST start

108:     assert_equal Time.utc(2006,4,2,3), twz2.time # twz is created for 3AM
109:     assert_equal Time.utc(2006,4,2,7), twz2.utc
110:     assert_equal true, twz2.dst?
111:     assert_equal 'EDT', twz2.zone
112:     twz3 = zone.local(2006,4,2,2,30) # 2:30AM does not exist because at 2AM, time springs forward to 3AM
113:     assert_equal Time.utc(2006,4,2,3,30), twz3.time # twz is created for 3:30AM
114:     assert_equal Time.utc(2006,4,2,7,30), twz3.utc
