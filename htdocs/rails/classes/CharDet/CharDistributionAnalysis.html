69:     def get_confidence
70:       # """return confidence based on existing data"""
71:       # if we didn't receive any character in our consideration range, return negative answer
72:       if @_mTotalChars <= 0
73:         return SURE_NO
74:       end
75: 
76:       if @_mTotalChars != @_mFreqChars
77:         r = @_mFreqChars / ((@_mTotalChars - @_mFreqChars) * @_mTypicalDistributionRatio)
78:         if r < SURE_YES
79:           return r
80:         end
81:       end
82: 
83:       # normalize confidence (we don't want to be 100% sure)
84:       return SURE_YES
85:     end
