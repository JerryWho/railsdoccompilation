197:     def get_order(aStr)
198:       # for sjis encoding, we are interested 
199:       #   first  byte range: 0x81 -- 0x9f , 0xe0 -- 0xfe
200:       #   second byte range: 0x40 -- 0x7e,  0x81 -- oxfe
201:       # no validation needed here. State machine has done that
202:       aStr = aStr[0..1].join if aStr.class == Array
203:       if (aStr[0..0] >= "\x81") and (aStr[0..0] <= "\x9F")
204:         order = 188 * (aStr[0] - 0x81)
205:       elsif (aStr[0..0] >= "\xE0") and (aStr[0..0] <= "\xEF")
206:         order = 188 * (aStr[0] - 0xE0 + 31)
207:       else
208:         return -1
209:       end
210:       order = order + aStr[1] - 0x40
211:       if aStr[1..1] > "\x7F"
212:         order =- 1
213:       end
214:       return order
215:     end
