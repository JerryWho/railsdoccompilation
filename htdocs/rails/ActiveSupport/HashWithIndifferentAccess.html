def self.new_from_hash_copying_default(hash)
  ActiveSupport::HashWithIndifferentAccess.new(hash).tap do |new_hash|
    new_hash.default = hash.default
  end
end

def default(key = nil)
  if key.is_a?(Symbol) && include?(key = key.to_s)
    self[key]
  else
    super
  end
end
