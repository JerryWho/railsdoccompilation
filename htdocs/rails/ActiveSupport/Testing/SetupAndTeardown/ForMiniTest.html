def run(runner)
  result = '.'
  begin
    _run_setup_callbacks do
      result = super
    end
  rescue Exception => e
    result = runner.puke(self.class, method_name, e)
  ensure
    begin
      _run_teardown_callbacks
    rescue Exception => e
      result = runner.puke(self.class, method_name, e)
    end
  end
  result
end
