def backend=(name)
  if name.is_a?(Module)
    @backend = name
  else
    require "active_support/json/backends/#{name.to_s.downcase}"
    @backend = ActiveSupport::JSON::Backends::const_get(name)
  end
  @parse_error = @backend::ParseError
end

def set_default_backend
  DECODERS.find do |name|
    begin
      self.backend = name
      true
    rescue LoadError
      # Try next decoder.
      false
    end
  end
end
