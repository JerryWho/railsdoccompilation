def render_template(event)
  message = "Rendered #{from_rails_root(event.payload[:identifier])}"
  message << " within #{from_rails_root(event.payload[:layout])}" if event.payload[:layout]
  message << (" (%.1fms)" % event.duration)
  info(message)
end
