def self.adapter_class
  case Webrat.configuration.mode
  when :rails
    RailsAdapter
  when :merb
    MerbAdapter
  when :rack
    RackAdapter
  when :rack_test
    warn("The :rack_test mode is deprecated. Please use :rack instead")
    require "webrat/rack"
    RackAdapter
  when :sinatra
    warn("The :sinatra mode is deprecated. Please use :rack instead")
    SinatraAdapter
  when :mechanize
    MechanizeAdapter
  else
    raise WebratError.new(<<-STRUnknown Webrat mode: #{Webrat.configuration.mode.inspect}

Please ensure you have a Webrat configuration block that specifies a mode
in your test_helper.rb, spec_helper.rb, or env.rb (for Cucumber).

This configure block supercedes the need to require "webrat/<framework>".

For example:

Webrat.configure do |config|
  config.mode = :rails
end
    STR
)
  end
end
