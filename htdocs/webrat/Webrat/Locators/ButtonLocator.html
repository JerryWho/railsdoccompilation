def button_element
  button_elements.detect do |element|
    @value.nil?             ||
    matches_id?(element)    ||
    matches_value?(element) ||
    matches_html?(element)  ||
    matches_alt?(element)
  end
end

def matches_id?(element)
  (@value.is_a?(Regexp) && element["id"] =~ @value) ||
  (!@value.is_a?(Regexp) && element["id"] == @value.to_s)
end
