def logger # :nodoc:
  case Webrat.configuration.mode
  when :rails
    defined?(RAILS_DEFAULT_LOGGER) ? RAILS_DEFAULT_LOGGER : nil
  when :merb
    ::Merb.logger
  else
    @logger ||= ::Logger.new("webrat.log")
  end
end
