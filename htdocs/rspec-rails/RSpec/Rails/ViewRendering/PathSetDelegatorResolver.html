def find_all(*args)
  path_set.find_all(*args).collect do |template|
    ::ActionView::Template.new(
      "",
      template.identifier,
      template.handler,
      {
        :virtual_path => template.virtual_path,
        :format => template.formats
      }
    )
  end
end
