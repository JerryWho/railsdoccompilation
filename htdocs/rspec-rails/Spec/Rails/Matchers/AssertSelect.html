def matches?(response_or_text, &block)
  @block = block if block

  if doc = doc_from(response_or_text)
    @args.unshift(doc)
  end

  begin
    @spec_scope.__send__(@selector_assertion, *@args, &@block)
    true
  rescue ::Test::Unit::AssertionFailedError => @error
    false
  end
end
