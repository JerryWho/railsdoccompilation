563:     if t.respond_to?(:to_datetime)      # 1.9 or later

576:       DateTime.new0(DateTime.jd_to_ajd(jd, fr, of), of, DateTime::ITALY)
577:     else

534:   def tz2of(str)
535:     /^(?:Z|(?:([+\-])(\d\d):(\d\d))?)$/ =~ str
536:     sign = $1
537:     hour = $2.to_i
538:     min = $3.to_i
539: 
540:     of = case sign
541:       when '+'
542:         of = +(hour.to_r * 60 + min) / 1440    # 24 * 60
543:       when '-'
544:         of = -(hour.to_r * 60 + min) / 1440    # 24 * 60
545:       else
546:         0
547:       end
548:     of
549:   end
