50:     ts = ''
51:     typedetails.each do |t|
52:       case t
53:       when 'CARRAY', 'VOID', 'UINT', 'RESULT', 'DECIMAL', 'I8', 'UI8' 
54: #         raise "Sorry type\"" + t + "\" not supported"
55:       ts << "\"??? NOT SUPPORTED TYPE:`#{t}'\""
56:       when 'USERDEFINED', 'Unknown Type 9'
57:         ts << 'VT_DISPATCH'
58:         break;
59:       when 'SAFEARRAY'
60:         ts << 'VT_ARRAY|'
61:       when 'PTR'
62:         ts << 'VT_BYREF|'
63:       when 'INT'
64:         ts << 'VT_I4'
65:       else
66:         if String === t
67:           ts << 'VT_' + t
68:         end
69:       end
70:     end
71:     if ts.empty?
72:       ts = 'VT_VARIANT'
73:     elsif ts[-1] == ?|
74:         ts += 'VT_VARIANT'
75:     end
76:     ts
77:   end
