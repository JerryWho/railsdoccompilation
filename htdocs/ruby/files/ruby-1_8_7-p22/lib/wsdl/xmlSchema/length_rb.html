require 'wsdl/info'


module WSDL
module XMLSchema


class Length < Info
  def initialize
    super
  end

  def parse_element(element)
    nil
  end

  def parse_attr(attr, value)
    case attr
    when ValueAttrName
      value.source
    end
  end
end


end
