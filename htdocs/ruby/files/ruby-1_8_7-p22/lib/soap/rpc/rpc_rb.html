module SOAP


module RPC
  ServerException = Mapping::MappedException

  def self.defined_methods(obj)
    if obj.is_a?(Module)
      obj.methods - Module.methods
    else
      obj.methods - Object.instance_methods(true)
    end
  end
end


end
